create database CRUD;
use CRUD;
create table USERS(id INT NOT NULL AUTO_INCREMENT, name VARCHAR(32) NOT NULL, password VARCHAR(32) NOT NULL, PRIMARY KEY (ID));
create table WORKOUTS(id INT NOT NULL AUTO_INCREMENT, user VARCHAR(32) NOT NULL, monday VARCHAR(255) NOT NULL, tuesday VARCHAR(255) NOT NULL, wednesday VARCHAR(255) NOT NULL, thursday VARCHAR(255) NOT NULL, friday VARCHAR(255) NOT NULL, saturday VARCHAR(255) NOT NULL, sunday VARCHAR(255) NOT NULL, PRIMARY KEY(ID));
